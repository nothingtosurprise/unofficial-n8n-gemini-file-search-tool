## Overview

This report documents the implementation of `GeminiFileSearchChatModel`, a custom LangChain chat model that integrates Gemini's File Search capability directly into the model layer.

## Implementation

### GeminiFileSearchChatModel Class

The following TypeScript/JavaScript code implements a custom LangChain `BaseChatModel` that wraps Gemini's `generateContent` API with the `fileSearch` tool pre-configured.

```typescript
/**
 * GeminiFileSearchChatModel
 *
 * A custom LangChain chat model that integrates Gemini's File Search
 * capability directly into the model, enabling single API call RAG queries.
 *
 * Usage in n8n LangChain Code Node:
 * - Mode: "Define a chat model to use with AI Agent"
 * - Returns: Instance of GeminiFileSearchChatModel
 */

const { BaseChatModel } = require('@langchain/core/language_models/chat_models');
const { AIMessage } = require('@langchain/core/messages');

// Configuration - Update these values for your setup
const CONFIG = {
  apiKey: '{{ $credentials.geminiApi.apiKey }}',  // Or hardcode for testing
  model: 'gemini-2.5-flash',
  storeNames: [
    'fileSearchStores/your-store-name'  // Replace with your store name(s)
  ],
  metadataFilter: '',  // Optional: e.g., 'category="docs" AND version>1.0'
  temperature: 0.7,
  maxOutputTokens: 2048,
};

class GeminiFileSearchChatModel extends BaseChatModel {
  constructor(config = {}) {
    super({});
    this.apiKey = config.apiKey || CONFIG.apiKey;
    this.modelName = config.model || CONFIG.model;
    this.storeNames = config.storeNames || CONFIG.storeNames;
    this.metadataFilter = config.metadataFilter || CONFIG.metadataFilter;
    this.temperature = config.temperature ?? CONFIG.temperature;
    this.maxOutputTokens = config.maxOutputTokens ?? CONFIG.maxOutputTokens;
  }

  _llmType() {
    return 'gemini-file-search';
  }

  /**
   * Convert LangChain messages to Gemini format
   */
  _convertMessages(messages) {
    const contents = [];

    for (const msg of messages) {
      const msgType = msg._getType();
      const content = typeof msg.content === 'string'
        ? msg.content
        : JSON.stringify(msg.content);

      if (msgType === 'system') {
        // Gemini handles system prompts as user+model exchange
        contents.push({
          role: 'user',
          parts: [{ text: content }],
        });
        contents.push({
          role: 'model',
          parts: [{ text: 'Understood. I will follow these instructions.' }],
        });
      } else if (msgType === 'human') {
        contents.push({
          role: 'user',
          parts: [{ text: content }],
        });
      } else if (msgType === 'ai') {
        contents.push({
          role: 'model',
          parts: [{ text: content }],
        });
      }
    }

    return contents;
  }

  /**
   * Call Gemini API with fileSearch tool
   */
  async _callGeminiApi(contents) {
    const url = `https://generativelanguage.googleapis.com/v1beta/models/${this.modelName}:generateContent?key=${this.apiKey}`;

    const body = {
      contents,
      tools: [{
        fileSearch: {
          fileSearchStoreNames: this.storeNames,
          ...(this.metadataFilter && { metadataFilter: this.metadataFilter }),
        },
      }],
      generationConfig: {
        temperature: this.temperature,
        maxOutputTokens: this.maxOutputTokens,
      },
    };

    const response = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(body),
    });

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`Gemini API error: ${response.status} - ${error}`);
    }

    return response.json();
  }

  /**
   * Main generation method - called by LangChain
   */
  async _generate(messages, options, runManager) {
    // Convert messages to Gemini format
    const contents = this._convertMessages(messages);

    // Call Gemini API
    const response = await this._callGeminiApi(contents);

    // Extract response data
    const candidate = response.candidates?.[0];
    const text = candidate?.content?.parts?.[0]?.text ?? '';
    const groundingMetadata = candidate?.groundingMetadata;

    // Create AIMessage with grounding metadata
    const aiMessage = new AIMessage({
      content: text,
      additional_kwargs: {
        groundingMetadata,
        groundingChunks: groundingMetadata?.groundingChunks,
        groundingSupports: groundingMetadata?.groundingSupports,
        usageMetadata: response.usageMetadata,
        finishReason: candidate?.finishReason,
      },
    });

    // Return in LangChain format
    return {
      generations: [{
        text,
        message: aiMessage,
        generationInfo: {
          finishReason: candidate?.finishReason,
          groundingChunks: groundingMetadata?.groundingChunks?.length ?? 0,
          totalTokens: response.usageMetadata?.totalTokenCount,
        },
      }],
      llmOutput: {
        tokenUsage: {
          promptTokens: response.usageMetadata?.promptTokenCount,
          completionTokens: response.usageMetadata?.candidatesTokenCount,
          totalTokens: response.usageMetadata?.totalTokenCount,
        },
      },
    };
  }

  /**
   * Bind tools - not supported, fileSearch is pre-configured
   */
  bindTools(tools) {
    console.warn('GeminiFileSearchChatModel: bindTools not supported, fileSearch is pre-configured');
    return this;
  }
}

// Export the model instance for n8n LangChain Code Node
return new GeminiFileSearchChatModel();
```

## Key Features

### 1. Single API Call
The model embeds the `fileSearch` tool directly in every API call, eliminating the need for a separate tool invocation.

### 2. Message Conversion
Properly converts LangChain message types to Gemini format:
- `SystemMessage` → User + Model exchange (Gemini limitation)
- `HumanMessage` → User role
- `AIMessage` → Model role

### 3. Grounding Metadata Preservation
All grounding metadata is preserved in `additional_kwargs`:
- `groundingChunks` - Source document references
- `groundingSupports` - Citation mappings
- `usageMetadata` - Token counts

### 4. Configuration Options

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `apiKey` | string | - | Gemini API key |
| `model` | string | `gemini-2.5-flash` | Model to use |
| `storeNames` | string[] | - | File Search Store names |
| `metadataFilter` | string | `''` | AIP-160 filter expression |
| `temperature` | number | `0.7` | Sampling temperature |
| `maxOutputTokens` | number | `2048` | Max response tokens |

### Accessing Grounding Metadata

After the AI Agent responds, grounding metadata is available in the output:

```javascript
// In a Code node after AI Agent
const response = $input.first().json;
const groundingChunks = response.additional_kwargs?.groundingChunks;

// Process citations
if (groundingChunks) {
  for (const chunk of groundingChunks) {
    console.log('Source:', chunk.retrievedContext?.title);
    console.log('Text:', chunk.retrievedContext?.text);
  }
}
```

## API Request Structure

The model sends requests in this format:

```json
{
  "contents": [
    {
      "role": "user",
      "parts": [{ "text": "System prompt here" }]
    },
    {
      "role": "model",
      "parts": [{ "text": "Understood. I will follow these instructions." }]
    },
    {
      "role": "user",
      "parts": [{ "text": "User question here" }]
    }
  ],
  "tools": [{
    "fileSearch": {
      "fileSearchStoreNames": ["fileSearchStores/my-store"],
      "metadataFilter": "category=\"docs\""
    }
  }],
  "generationConfig": {
    "temperature": 0.7,
    "maxOutputTokens": 2048
  }
}
```

## Error Handling

The model handles common errors:

| Error | Cause | Solution |
|-------|-------|----------|
| 401 Unauthorized | Invalid API key | Check API key |
| 404 Not Found | Invalid store name | Verify store exists |
| 400 Bad Request | Invalid filter syntax | Check metadataFilter |
| 429 Too Many Requests | Rate limit exceeded | Add retry logic |

## Limitations

1. **No Streaming**: Initial version does not support streaming responses
2. **No Tool Binding**: `bindTools()` is not supported - fileSearch is pre-configured
3. **System Prompt Workaround**: Gemini doesn't support system prompts natively; converted to user+model exchange
